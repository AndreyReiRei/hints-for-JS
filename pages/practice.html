class TodoApp {
    constructor() {
        this.tasks = JSON.parse(localStorage.getItem('todoTasks')) || [];
        this.currentFilter = 'all';
        this.init();
    }

    init() {
        this.render();
        this.setupEventListeners();
    }

    render() {
        const filteredTasks = this.getFilteredTasks();
        this.renderTaskList(filteredTasks);
        this.renderStats();
        this.saveToLocalStorage();
    }

    getFilteredTasks() {
        switch(this.currentFilter) {
            case 'active':
                return this.tasks.filter(task => !task.completed);
            case 'completed':
                return this.tasks.filter(task => task.completed);
            default:
                return this.tasks;
        }
    }

    renderTaskList(tasks) {
        const taskList = document.getElementById('taskList');
        taskList.innerHTML = tasks.map((task, index) => `
            <li class="task-item ${task.completed ? 'completed' : ''}">
                <input type="checkbox" 
                       class="task-checkbox" 
                       ${task.completed ? 'checked' : ''}
                       onchange="todoApp.toggleTask(${index})">
                <span class="task-text">${task.text}</span>
                <button class="delete-btn" 
                        onclick="todoApp.deleteTask(${index})">
                    –£–¥–∞–ª–∏—Ç—å
                </button>
            </li>
        `).join('');
    }

    renderStats() {
        const total = this.tasks.length;
        const completed = this.tasks.filter(t => t.completed).length;
        const active = total - completed;
        
        document.getElementById('taskCount').textContent = 
            `${active} –∞–∫—Ç–∏–≤–Ω—ã—Ö, ${completed} –≤—ã–ø–æ–ª–Ω–µ–Ω–Ω—ã—Ö`;
    }

    addTask(text) {
        if (text.trim()) {
            this.tasks.push({
                text: text.trim(),
                completed: false,
                createdAt: new Date()
            });
            this.render();
        }
    }

    toggleTask(index) {
        this.tasks[index].completed = !this.tasks[index].completed;
        this.render();
    }

    deleteTask(index) {
        this.tasks.splice(index, 1);
        this.render();
    }

    clearCompleted() {
        this.tasks = this.tasks.filter(task => !task.completed);
        this.render();
    }

    setFilter(filter) {
        this.currentFilter = filter;
        this.render();
    }

    saveToLocalStorage() {
        localStorage.setItem('todoTasks', JSON.stringify(this.tasks));
    }

    setupEventListeners() {
        document.getElementById('addBtn').addEventListener('click', () => {
            const input = document.getElementById('taskInput');
            this.addTask(input.value);
            input.value = '';
        });

        document.getElementById('taskInput').addEventListener('keypress', (e) => {
            if (e.key === 'Enter') {
                this.addTask(e.target.value);
                e.target.value = '';
            }
        });

        document.querySelectorAll('.filter-btn').forEach(btn => {
            btn.addEventListener('click', (e) => {
                document.querySelectorAll('.filter-btn').forEach(b => 
                    b.classList.remove('active')
                );
                e.target.classList.add('active');
                this.setFilter(e.target.dataset.filter);
            });
        });

        document.getElementById('clearBtn').addEventListener('click', () => {
            this.clearCompleted();
        });
    }
}

    async getWeatherByCity(city) {
        try {
            const url = `${this.baseUrl}?q=${city}&appid=${this.apiKey}&units=metric&lang=ru`;
            const response = await fetch(url);
            
            if (!response.ok) {
                throw new Error('–ì–æ—Ä–æ–¥ –Ω–µ –Ω–∞–π–¥–µ–Ω');
            }
            
            const data = await response.json();
            this.addToHistory(city);

        } catch (error) {
            throw new Error(`–û—à–∏–±–∫–∞: ${error.message}`);
        }
    }

    async getWeatherByLocation(lat, lon) {
        try {
            const url = `${this.baseUrl}?lat=${lat}&lon=${lon}&appid=${this.apiKey}&units=metric&lang=ru`;
            const response = await fetch(url);
            const data = await response.json();
            return this.formatWeatherData(data);
            
        } catch (error) {
            throw new Error('–ù–µ —É–¥–∞–ª–æ—Å—å –ø–æ–ª—É—á–∏—Ç—å –ø–æ–≥–æ–¥—É –ø–æ –º–µ—Å—Ç–æ–ø–æ–ª–æ–∂–µ–Ω–∏—é');
        }

    formatWeatherData(data) {
        return {
            city: data.name,
            country: data.sys.country,
            temperature: Math.round(data.main.temp),
            feelsLike: Math.round(data.main.feels_like),
            humidity: data.main.humidity,
            pressure: data.main.pressure,
            windSpeed: data.wind.speed,
            description: data.weather[0].description,
            icon: data.weather[0].icon,
            sunrise: new Date(data.sys.sunrise * 1000),
            sunset: new Date(data.sys.sunset * 1000)
        };
    }

    addToHistory(city) {
        const entry = {
            city,
            timestamp: new Date().toLocaleString()
        };
        
        this.history.unshift(entry);
        if (this.history.length > 5) {
            this.history.pop();
        }
        
        localStorage.setItem('weatherHistory', JSON.stringify(this.history));
        this.renderHistory();
    }

    renderHistory() {

        historyElement.innerHTML = this.history.map(entry => `

class QuizGame {
    constructor() {
        this.questions = [];
        this.currentQuestionIndex = 0;
        this.score = 0;
        this.timeLeft = 30;
        this.timer = null;
        this.selectedQuestions = [];
    }

    startQuiz(questionCount) {
        this.selectedQuestions = this.getRandomQuestions(questionCount);
        this.currentQuestionIndex = 0;
        this.score = 0;
        this.showQuestion();
        this.startTimer();
    }

    getRandomQuestions(count) {
        // –ü–µ—Ä–µ–º–µ—à–∏–≤–∞–µ–º –≤–æ–ø—Ä–æ—Å—ã –∏ –±–µ—Ä–µ–º –Ω—É–∂–Ω–æ–µ –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ
        return [...quizQuestions]
            .sort(() => Math.random() - 0.5)
            .slice(0, count);
    }

    showQuestion() {
        const question = this.selectedQuestions[this.currentQuestionIndex];
        
        // –û–±–Ω–æ–≤–ª—è–µ–º –∏–Ω—Ç–µ—Ä—Ñ–µ–π—Å
        document.getElementById('questionText').textContent = question.question;
        document.getElementById('currentQuestion').textContent = 
            this.currentQuestionIndex + 1;
        
        // –ì–µ–Ω–µ—Ä–∏—Ä—É–µ–º –≤–∞—Ä–∏–∞–Ω—Ç—ã –æ—Ç–≤–µ—Ç–æ–≤
        const answersContainer = document.getElementById('answersContainer');
        answersContainer.innerHTML = question.answers.map((answer, index) => `
            <button class="answer-btn" 
                    onclick="quiz.selectAnswer(${index})"
                    data-answer="${index}">
                ${answer}
            </button>
        `).join('');
        
        // –°–±—Ä–∞—Å—ã–≤–∞–µ–º —Å–æ—Å—Ç–æ—è–Ω–∏–µ
        document.getElementById('nextBtn').disabled = true;
        this.resetTimer();
    }

    selectAnswer(answerIndex) {
        const question = this.selectedQuestions[this.currentQuestionIndex];
        const isCorrect = answerIndex === question.correctAnswer;
        
        // –í–∏–∑—É–∞–ª—å–Ω–∞—è –æ–±—Ä–∞—Ç–Ω–∞—è —Å–≤—è–∑—å
        const answerButtons = document.querySelectorAll('.answer-btn');
        answerButtons.forEach((btn, index) => {
            btn.classList.remove('selected', 'correct', 'wrong');
            
            if (index === answerIndex) {
                btn.classList.add('selected');
                btn.classList.add(isCorrect ? 'correct' : 'wrong');
            }
            
            if (index === question.correctAnswer) {
                btn.classList.add('correct');
            }
            
            btn.disabled = true;
        });

        // –û–±–Ω–æ–≤–ª—è–µ–º —Å—á–µ—Ç
        if (isCorrect) {
            this.score += this.calculatePoints(question.difficulty);
            document.getElementById('score').textContent = this.score;
        }

        // –ü–æ–∫–∞–∑—ã–≤–∞–µ–º –æ–±—ä—è—Å–Ω–µ–Ω–∏–µ
        this.showExplanation(question.explanation);

        // –†–∞–∑—Ä–µ—à–∞–µ–º –ø–µ—Ä–µ–π—Ç–∏ –∫ —Å–ª–µ–¥—É—é—â–µ–º—É –≤–æ–ø—Ä–æ—Å—É
        document.getElementById('nextBtn').disabled = false;
        clearInterval(this.timer);
    }

    calculatePoints(difficulty) {
        const points = {
            easy: 10,
            medium: 20,
            hard: 30
        };
        return points[difficulty] || 10;
    }

    nextQuestion() {
        this.currentQuestionIndex++;
        
        if (this.currentQuestionIndex < this.selectedQuestions.length) {
            this.showQuestion();
            this.startTimer();
        } else {
            this.finishQuiz();
        }
    }

    finishQuiz() {
        clearInterval(this.timer);
        
        const percentage = Math.round(
            (this.score / (this.selectedQuestions.length * 20)) * 100
        );
        
        let message = '';
        if (percentage >= 80) message = '–û—Ç–ª–∏—á–Ω–æ! –í—ã —ç–∫—Å–ø–µ—Ä—Ç JavaScript!';
        else if (percentage >= 60) message = '–•–æ—Ä–æ—à–æ! –ü—Ä–æ–¥–æ–ª–∂–∞–π—Ç–µ –ø—Ä–∞–∫—Ç–∏–∫–æ–≤–∞—Ç—å—Å—è!';
        else message = '–ï—Å—Ç—å –Ω–∞–¥ —á–µ–º –ø–æ—Ä–∞–±–æ—Ç–∞—Ç—å!';
        
        document.getElementById('quizResult').innerHTML = `
            <h3>–í–∏–∫—Ç–æ—Ä–∏–Ω–∞ –∑–∞–≤–µ—Ä—à–µ–Ω–∞!</h3>
            <div class="result-stats">
                <p>–í–∞—à —Ä–µ–∑—É–ª—å—Ç–∞—Ç: <strong>${this.score} –æ—á–∫–æ–≤</strong></p>
                <p>–ü—Ä–∞–≤–∏–ª—å–Ω—ã—Ö –æ—Ç–≤–µ—Ç–æ–≤: ${Math.round(this.score / 20)} –∏–∑ ${this.selectedQuestions.length}</p>
                <p>–ü—Ä–æ—Ü–µ–Ω—Ç —É—Å–ø–µ—Ö–∞: ${percentage}%</p>
                <p>${message}</p>
            </div>
            <button class="btn btn-primary" onclick="quiz.restartQuiz()">
                <i class="fas fa-redo"></i> –ü—Ä–æ–π—Ç–∏ –µ—â–µ —Ä–∞–∑
            </button>
        `;
        
        document.getElementById('quizContainer').classList.add('hidden');
        document.getElementById('quizResult').classList.remove('hidden');
    }

    startTimer() {
        this.timeLeft = parseInt(document.getElementById('timePerQuestion').value);
        document.getElementById('timer').textContent = this.timeLeft;
        
        this.timer = setInterval(() => {
            this.timeLeft--;
            document.getElementById('timer').textContent = this.timeLeft;
            
            if (this.timeLeft <= 0) {
                clearInterval(this.timer);
                this.skipQuestion();
            }
        }, 1000);
    }

    skipQuestion() {
        clearInterval(this.timer);
        this.showExplanation('–í—Ä–µ–º—è –≤—ã—à–ª–æ!');
        document.getElementById('nextBtn').disabled = false;
    }

    showExplanation(text) {
        // –ü–æ–∫–∞–∑–∞—Ç—å –æ–±—ä—è—Å–Ω–µ–Ω–∏–µ –ø–æ–¥ –≤–∞—Ä–∏–∞–Ω—Ç–∞–º–∏ –æ—Ç–≤–µ—Ç–æ–≤
        const explanationDiv = document.createElement('div');
        explanationDiv.className = 'question-explanation';
        explanationDiv.innerHTML = `<p><i class="fas fa-info-circle"></i> ${text}</p>`;
        
        const answersContainer = document.getElementById('answersContainer');
        answersContainer.appendChild(explanationDiv);
    }
}

class NotesApp {
    constructor() {
        this.notes = JSON.parse(localStorage.getItem('notes')) || [];
        this.tags = new Set();
        this.currentFilter = 'all';
        this.init();
    }

    init() {
        this.extractTags();
        this.renderNotes();
        this.renderTags();
    }

    addNote(title, content, tags = []) {
        const note = {
            id: Date.now(),
            title,
            content,
            tags,
            createdAt: new Date(),
            updatedAt: new Date(),
            color: this.getRandomColor(),
            pinned: false
        };

        this.notes.unshift(note);
        this.saveAndRender();
    }

    editNote(id, updates) {
        const noteIndex = this.notes.findIndex(note => note.id === id);
        if (noteIndex !== -1) {
            this.notes[noteIndex] = {
                ...this.notes[noteIndex],
                ...updates,
                updatedAt: new Date()
            };
            this.saveAndRender();
        }
    }

    deleteNote(id) {
        this.notes = this.notes.filter(note => note.id !== id);
        this.saveAndRender();
    }

    togglePin(id) {
        const note = this.notes.find(n => n.id === id);
        if (note) {
            note.pinned = !note.pinned;
            this.notes.sort((a, b) => b.pinned - a.pinned);
            this.saveAndRender();
        }
    }

    extractTags() {
        this.tags.clear();
        this.notes.forEach(note => {
            note.tags.forEach(tag => this.tags.add(tag));
        });
    }

    filterNotesByTag(tag) {
        if (tag === 'all') {
            return this.notes;
        }
        return this.notes.filter(note => note.tags.includes(tag));
    }

    searchNotes(query) {
        const searchTerm = query.toLowerCase();
        return this.notes.filter(note => 
            note.title.toLowerCase().includes(searchTerm) ||
            note.content.toLowerCase().includes(searchTerm) ||
            note.tags.some(tag => tag.toLowerCase().includes(searchTerm))
        );
    }

    saveAndRender() {
        this.extractTags();
        localStorage.setItem('notes', JSON.stringify(this.notes));
        this.renderNotes();
        this.renderTags();
    }

    getRandomColor() {
        const colors = [
            '#FFEBEE', '#F3E5F5', '#E8EAF6', 
            '#E3F2FD', '#E8F5E8', '#FFF8E1'
        ];
        return colors[Math.floor(Math.random() * colors.length)];
    }

    renderNotes() {
        const notesElement = document.getElementById('notesContainer');
        const filteredNotes = this.currentFilter === 'all' 
            ? this.notes 
            : this.filterNotesByTag(this.currentFilter);

        notesElement.innerHTML = filteredNotes.map(note => `
            <div class="note-card" style="background: ${note.color}">
                <div class="note-header">
                    <h4 class="note-title">${note.pinned ? 'üìå ' : ''}${note.title}</h4>
                    <div class="note-actions">
                        <button onclick="notesApp.togglePin(${note.id})" 
                                class="btn-icon ${note.pinned ? 'pinned' : ''}">
                            <i class="fas fa-thumbtack"></i>
                        </button>
                        <button onclick="notesApp.editNoteModal(${note.id})" 
                                class="btn-icon">
                            <i class="fas fa-edit"></i>
                        </button>
                        <button onclick="notesApp.deleteNote(${note.id})" 
                                class="btn-icon delete">
                            <i class="fas fa-trash"></i>
                        </button>
                    </div>
                </div>
                <div class="note-content">
                    ${this.formatNoteContent(note.content)}
                </div>
                <div class="note-footer">
                    <div class="note-tags">
                        ${note.tags.map(tag => `
                            <span class="note-tag" onclick="notesApp.filterByTag('${tag}')">
                                #${tag}
                            </span>
                        `).join('')}
                    </div>
                    <div class="note-date">
                        ${new Date(note.updatedAt).toLocaleDateString()}
                    </div>
                </div>
            </div>
        `).join('');
    }

    formatNoteContent(content) {
        // –§–æ—Ä–º–∞—Ç–∏—Ä–æ–≤–∞–Ω–∏–µ —Ç–µ–∫—Å—Ç–∞: —Å—Å—ã–ª–∫–∏, –∂–∏—Ä–Ω—ã–π —Ç–µ–∫—Å—Ç –∏ —Ç.–¥.
        return content
            .replace(/\n/g, '<br>')
            .replace(/\*\*(.*?)\*\*/g, '<strong>$1</strong>')
            .replace(/\*(.*?)\*/g, '<em>$1</em>')
            .replace(/`(.*?)`/g, '<code>$1</code>');
    }

    renderTags() {
        const tagsElement = document.getElementById('tagsContainer');
        tagsElement.innerHTML = `
            <button class="tag-btn ${this.currentFilter === 'all' ? 'active' : ''}" 
                    onclick="notesApp.setFilter('all')">
                –í—Å–µ –∑–∞–º–µ—Ç–∫–∏ (${this.notes.length})
            </button>
            ${Array.from(this.tags).map(tag => {
                const count = this.notes.filter(n => n.tags.includes(tag)).length;
                return `
                    <button class="tag-btn ${this.currentFilter === tag ? 'active' : ''}" 
                            onclick="notesApp.setFilter('${tag}')">
                        #${tag} (${count})
                    </button>
                `;
            }).join('')}
        `;
    }

    setFilter(filter) {
        this.currentFilter = filter;
        this.renderNotes();
    }
}

// –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è
const notesApp = new NotesApp();
