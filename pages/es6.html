let и const

let и const — современные способы объявления переменных, которые решают проблемы var.

Ключевое слово let

// Блочная область видимости
{
    let x = 10;
    console.log(x); // 10
}
// console.log(x); // ❌ Ошибка: x is not defined

// Отличие от var (поднятие)
console.log(y); // undefined (var всплывает)
var y = 5;

// console.log(z); // ❌ Ошибка: Cannot access 'z' before initialization
let z = 5;

// Повторное объявление
var a = 1;
var a = 2; // ✅ Работает

let b = 1;
// let b = 2; // ❌ Ошибка: Identifier 'b' has already been declared

// Циклы
for (let i = 0; i < 3; i++) {
    setTimeout(() => console.log(i), 100); // 0, 1, 2
}

// Для сравнения с var:
for (var j = 0; j < 3; j++) {
    setTimeout(() => console.log(j), 100); // 3, 3, 3
}

Сравнение var, let, const

Критерий var let const
Область видимости Функция Блок Блок
Поднятие (hoisting) ✅ С инициализацией ✅ Без инициализации ✅ Без инициализации
Повторное объявление ✅ Разрешено ❌ Запрещено ❌ Запрещено
Изменение значения ✅ Разрешено ✅ Разрешено ❌ Запрещено
Инициализация при объявлении ❌ Не требуется ❌ Не требуется ✅ Обязательна

Правила использования:

  1. Всегда используйте const по умолчанию
  2. Используйте let только когда значение будет меняться
  3. Избегайте var в современном коде
  4. Для объектов и массивов используйте const — это защищает от случайного переназначения

Стрелочные функции

Стрелочные функции предоставляют краткий синтаксис и особое поведение ключевого слова this.

Базовый синтаксис

// Обычная функция
function add(a, b) {
    return a + b;
}

// Стрелочная функция
const addArrow = (a, b) => {
    return a + b;
};

// Еще короче (неявный return)
const addShort = (a, b) => a + b;

// Без аргументов
const greet = () => "Привет!";

// Один аргумент (скобки optional)
const square = x => x * x;

// Возврат объекта
const createUser = (name, age) => ({
    name: name,
    age: age
});

console.log(add(2, 3)); // 5
console.log(addArrow(2, 3)); // 5
console.log(addShort(2, 3)); // 5
console.log(square(4)); // 16

this в стрелочных функциях

// Обычная функция
const obj1 = {
    value: 10,
    getValue: function() {
        return this.value;
    }
};

console.log(obj1.getValue()); // 10

// Проблема с this
const obj2 = {
    value: 10,
    getValue: function() {
        setTimeout(function() {
            console.log(this.value); // undefined
        }, 100);
    }
};

// Решение со стрелочной функцией
const obj3 = {
    value: 10,
    getValue: function() {
        setTimeout(() => {
            console.log(this.value); // 10
        }, 100);
    }
};

// Стрелочные функции НЕ ИМЕЮТ своего this
const obj4 = {
    value: 10,
    getValue: () => {
        console.log(this.value); // undefined
    }
};

obj2.getValue();
obj3.getValue();
obj4.getValue();

Методы массивов

const numbers = [1, 2, 3, 4, 5];

// map
const doubled = numbers.map(n => n * 2);
console.log(doubled); // [2, 4, 6, 8, 10]

// filter
const evens = numbers.filter(n => n % 2 === 0);
console.log(evens); // [2, 4]

// reduce
const sum = numbers.reduce((acc, n) => acc + n, 0);
console.log(sum); // 15

// Цепочка методов
const result = numbers
    .filter(n => n > 2)
    .map(n => n * 3)
    .reduce((acc, n) => acc + n, 0);

console.log(result); // 36 (3*3 + 4*3 + 5*3)

// find
const firstEven = numbers.find(n => n % 2 === 0);
console.log(firstEven); // 2

// some/every
const hasNegative = numbers.some(n => n < 0);
const allPositive = numbers.every(n => n > 0);
console.log(hasNegative, allPositive); // false true

Ограничения стрелочных функций:

  • Не имеют своего this, arguments, super, new.target
  • Не могут быть использованы как конструкторы (нельзя использовать с new)
  • Не имеют свойства prototype
  • Не подходят для методов объектов, где нужен собственный this
  • Не могут быть использованы как генераторы

// Tagged template - функция, обрабатывающая шаблонную строку
function highlight(strings, ...values) {
    let result = '';
    strings.forEach((str, i) => {
        result += str;
        if (i < values.length) {
            result += `${values[i]}`;
        }
    });
    return result;
}

const name = "Анна";
const score = 95;
const taggedResult = highlight`Студент ${name} получил ${score} баллов`;
console.log(taggedResult);

// Безопасность: экранирование HTML
function safeHtml(strings, ...values) {
    return strings.reduce((result, str, i) => {
        const value = values[i] ? 
            String(values[i])
                .replace(/&/g, '&')
                .replace(//g, '>')
                .replace(/"/g, '"')
                .replace(/'/g, ''') : '';
        return result + str + value;
    }, '');
}

const userInput = '';
const safeResult = safeHtml`
${userInput}
`;
console.log("Безопасный HTML:", safeResult);

// Форматирование чисел
function currency(strings, ...values) {
    return strings.reduce((result, str, i) => {
        let value = values[i];
        if (typeof value === 'number') {
            value = value.toLocaleString('ru-RU', {
                style: 'currency',
                currency: 'RUB'
            });
        }
        return result + str + (value || '');
    }, '');
}

const price = 1500;
const discount = 200;
console.log(currency`Цена: ${price}, скидка: ${discount}, итого: ${price - discount}`);

// Локализация
function l10n(strings, ...values) {
    const translations = {
        'Hello': 'Привет',
        'World': 'Мир',
        'welcome': 'добро пожаловать'
    };
    
    return strings.reduce((result, str, i) => {
        const value = values[i] ? translations[values[i]] || values[i] : '';
        return result + translations[str.trim()] || str + value;
    }, '');
}

console.log(l10n`Hello ${'World'}! welcome to our site.`);

Деструктуризация

Деструктуризация позволяет извлекать значения из объектов и массивов в отдельные переменные.

1

Деструктуризация объектов

const user = {
    name: "Анна",
    age: 25,
    email: "anna@example.com",
    address: {
        city: "Москва",
        street: "Тверская"
    }
};

// Базовый синтаксис
const { name, age } = user;
console.log(name, age); // Анна 25

// Новые имена переменных
const { name: userName, age: userAge } = user;
console.log(userName, userAge); // Анна 25

// Значения по умолчанию
const { phone = "не указан" } = user;
console.log(phone); // не указан

// Вложенная деструктуризация
const { address: { city, street } } = user;
console.log(city, street); // Москва Тверская

// Остальные свойства
const { email, ...rest } = user;
console.log(email); // anna@example.com
console.log(rest); // {name: "Анна", age: 25, address: {...}}

// В параметрах функции
function printUser({ name, age, city = "не указан" }) {
    console.log(`${name}, ${age} лет, город: ${city}`);
}
printUser(user);

// Возврат из функции
function getUser() {
    return { id: 1, login: "admin", role: "administrator" };
}
const { login, role } = getUser();
console.log(login, role); // admin administrator

Spread и Rest операторы

Оператор ... (spread/rest) позволяет работать с коллекциями данных более гибко.

// ===== МАССИВЫ =====
const arr1 = [1, 2, 3];
const arr2 = [4, 5, 6];

// Копирование массива
const copy = [...arr1];
console.log(copy); // [1, 2, 3]

// Объединение массивов
const merged = [...arr1, ...arr2];
console.log(merged); // [1, 2, 3, 4, 5, 6]

// Добавление элементов
const withExtra = [0, ...arr1, 4];
console.log(withExtra); // [0, 1, 2, 3, 4]

// Преобразование строки
const str = "hello";
const chars = [...str];
console.log(chars); // ['h', 'e', 'l', 'l', 'o']

// Преобразование NodeList/Set
const set = new Set([1, 2, 3, 2, 1]);
const arrayFromSet = [...set];
console.log(arrayFromSet); // [1, 2, 3]

// ===== ОБЪЕКТЫ (ES2018) =====
const obj1 = { a: 1, b: 2 };
const obj2 = { c: 3, d: 4 };

// Копирование объекта
const objCopy = { ...obj1 };
console.log(objCopy); // {a: 1, b: 2}

// Объединение объектов
const objMerged = { ...obj1, ...obj2 };
console.log(objMerged); // {a: 1, b: 2, c: 3, d: 4}

// Перезапись свойств
const updated = { ...obj1, b: 20 };
console.log(updated); // {a: 1, b: 20}

// Добавление свойств
const withNew = { ...obj1, e: 5 };
console.log(withNew); // {a: 1, b: 2, e: 5}

// Порядок важен
const result1 = { x: 1, ...obj1, x: 2 };
console.log(result1); // {x: 2, a: 1, b: 2}

function processData(...args) {
    const numbers = args.filter(arg => typeof arg === 'number');
    const strings = args.filter(arg => typeof arg === 'string');
    const others = args.filter(arg => 
        typeof arg !== 'number' && typeof arg !== 'string'

}

// Полифил для bind (пример)
function bindPolyfill(fn, context, ...boundArgs) {
    return function(...args) {
        return fn.apply(context, [...boundArgs, ...args]);
    };
}

const obj = { value: 10 };
function multiply(a, b) {
    return this.value * a * b;
}

const boundMultiply = bindPolyfill(multiply, obj, 2);
console.log(boundMultiply(3)); // 60 (10 * 2 * 3)

// Именованный экспорт (несколько)
export const PI = 3.14159;

export function add(a, b) {
    return a + b;
}

export function multiply(a, b) {
    return a * b;
}

// Экспорт существующих переменных
const subtract = (a, b) => a - b;
const divide = (a, b) => a / b;

export { subtract, divide };

// Переименование при экспорте
const secretAlgorithm = () => "секрет";
export { secretAlgorithm as publicAlgorithm };

// Групповой экспорт
export { 
    PI as piConstant, 
    add as sum, 
    multiply as product 
};

// Экспорт по умолчанию (только один!)
export default function calculate(operation, a, b) {
    switch(operation) {
        case '+': return add(a, b);
        case '-': return subtract(a, b);
        case '*': return multiply(a, b);
        case '/': return divide(a, b);
        default: return NaN;
    }
}

// Можно экспортировать классы
export class Calculator {
    constructor() {
        this.result = 0;
    }
    
    add(x) {
        this.result += x;
        return this;
    }
    
    getResult() {
        return this.result;
    }
}

// Решение: Конвертер валют на ES6+

// ===== МОДУЛЬ: КОНСТАНТЫ И УТИЛИТЫ =====
const CURRENCY_SYMBOLS = {
    USD: '$',
    EUR: '€',
    RUB: '₽',
    JPY: '¥',
    GBP: '£'
};

const API_ENDPOINTS = {
    EXCHANGE_RATE: 'https://api.exchangerate-api.com/v4/latest/'
};

// ===== МОДУЛЬ: СЕРВИС КУРСОВ ВАЛЮТ =====
class ExchangeRateService {
    constructor() {
        this.cache = new Map();
        this.cacheDuration = 5 * 60 * 1000; // 5 минут
    }
    
    async fetchRate(baseCurrency = 'USD') {
        // Проверяем кэш
        const cached = this.cache.get(baseCurrency);
        if (cached && Date.now() - cached.timestamp < this.cacheDuration) {
            console.log(`Используем кэшированный курс для ${baseCurrency}`);
            return cached.rates;
        }
        
        try {
            const response = await fetch(`${API_ENDPOINTS.EXCHANGE_RATE}${baseCurrency}`);
            if (!response.ok) throw new Error(`HTTP ${response.status}`);
            
            const data = await response.json();
            const { rates, date: timestamp } = data;
            
            // Сохраняем в кэш
            this.cache.set(baseCurrency, {
                rates,
                timestamp: Date.now()
            });
            
            return rates;
        } catch (error) {
            console.error('Ошибка загрузки курсов:', error);
            return null;
        }
    }
    
    getCachedRates(baseCurrency) {
        const cached = this.cache.get(baseCurrency);
        return cached ? cached.rates : null;
    }
    
    clearCache() {
        this.cache.clear();
        console.log('Кэш очищен');
    }
}

// ===== МОДУЛЬ: КОНВЕРТЕР =====
class CurrencyConverter {
    constructor(rateService) {
        this.rateService = rateService;
        this.baseCurrency = 'USD';
        this.rates = {};
    }
    
    async initialize() {
        this.rates = await this.rateService.fetchRate(this.baseCurrency);
        if (!this.rates) {
            throw new Error('Не удалось загрузить курсы валют');
        }
        return this;
    }
    
    convert({ amount, from, to }) {
        if (!this.rates[from] || !this.rates[to]) {
            throw new Error(`Неизвестная валюта: ${from} или ${to}`);
        }
        
        // Конвертируем через базовую валюту
        const amountInBase = amount / this.rates[from];
        const convertedAmount = amountInBase * this.rates[to];
        
        return {
            amount: convertedAmount,
            from,
            to,
            rate: this.rates[to] / this.rates[from],
            timestamp: Date.now()
        };
    }
    
    format({ amount, to }, locale = 'ru-RU') {
        const symbol = CURRENCY_SYMBOLS[to] || to;
        const formatter = new Intl.NumberFormat(locale, {
            style: 'currency',
            currency: to,
            minimumFractionDigits: 2,
            maximumFractionDigits: 2
        });
        
        return formatter.format(amount);
    }
    
    getAvailableCurrencies() {
        return Object.keys(this.rates).sort();
    }
    
    getExchangeRate(from, to) {
        return this.rates[to] / this.rates[from];
    }
}

// ===== МОДУЛЬ: ИСТОРИЯ ОПЕРАЦИЙ =====
class ConversionHistory {
    constructor() {
        this.history = new Set();
        this.maxSize = 10;
    }
    
    add(conversion) {
        const record = {
            ...conversion,
            id: Date.now() + Math.random().toString(36).substr(2, 9),
            formatted: this.formatRecord(conversion)
        };
        
        this.history.add(record);
        
        // Ограничиваем размер истории
        if (this.history.size > this.maxSize) {
            const oldest = Array.from(this.history)
                .sort((a, b) => a.timestamp - b.timestamp)[0];
            this.history.delete(oldest);
        }
        
        return record;
    }
    
    formatRecord({ amount, from, to, rate }) {
        return `${amount} ${from} → ${(amount * rate).toFixed(2)} ${to} (курс: ${rate.toFixed(4)})`;
    }
    
    getAll() {
        return Array.from(this.history)
            .sort((a, b) => b.timestamp - a.timestamp);
    }
    
    clear() {
        this.history.clear();
    }
    
    exportToJSON() {
        return JSON.stringify(this.getAll(), null, 2);
    }
}

// ===== МОДУЛЬ: ИНТЕРФЕЙС ПОЛЬЗОВАТЕЛЯ =====
class CurrencyConverterUI {
    constructor(converter, history) {
        this.converter = converter;
        this.history = history;
        this.initializeUI();
    }
    
    initializeUI() {
        this.renderCurrencySelectors();
        this.setupEventListeners();
        this.updateDisplay();
    }
    
    renderCurrencySelectors() {
        const currencies = this.converter.getAvailableCurrencies();
        const fromSelect = document.getElementById('fromCurrency');
        const toSelect = document.getElementById('toCurrency');
        
        currencies.forEach(currency => {
            const option = document.createElement('option');
            option.value = currency;
            option.textContent = `${currency} ${CURRENCY_SYMBOLS[currency] || ''}`;
            
            fromSelect.appendChild(option.cloneNode(true));
            toSelect.appendChild(option);
        });
        
        fromSelect.value = 'USD';
        toSelect.value = 'EUR';
    }
    
    setupEventListeners() {
        document.getElementById('convertBtn').addEventListener('click', 
            () => this.handleConvert()
        );
        
        document.getElementById('swapBtn').addEventListener('click', 
            () => this.swapCurrencies()
        );
        
        document.getElementById('clearHistoryBtn').addEventListener('click', 
            () => this.clearHistory()
        );
    }
    
    async handleConvert() {
        const amount = parseFloat(document.getElementById('amount').value);
        const from = document.getElementById('fromCurrency').value;
        const to = document.getElementById('toCurrency').value;
        
        if (!amount || amount <= 0) {
            this.showError('Введите корректную сумму');
            return;
        }
        
        try {
            const result = this.converter.convert({ amount, from, to });
            const formatted = this.converter.format({ amount: result.amount, to });
            
            // Добавляем в историю
            this.history.add({
                ...result,
                originalAmount: amount
            });
            
            // Обновляем отображение
            this.updateResult(formatted, result.rate);
            this.updateHistory();
            
        } catch (error) {
            this.showError(error.message);
        }
    }
    
    swapCurrencies() {
        const fromSelect = document.getElementById('fromCurrency');
        const toSelect = document.getElementById('toCurrency');
        
        [fromSelect.value, toSelect.value] = [toSelect.value, fromSelect.value];
    }
    
    updateResult(formattedAmount, rate) {
        document.getElementById('result').textContent = formattedAmount;
        document.getElementById('rate').textContent = rate.toFixed(4);
    }
    
    updateHistory() {
        const historyList = document.getElementById('historyList');
        historyList.innerHTML = '';
        
        this.history.getAll().forEach(record => {
            const li = document.createElement('li');
            li.className = 'history-item';
            li.innerHTML = `
                ${record.formatted}
                ${new Date(record.timestamp).toLocaleTimeString()}
            `;
            historyList.appendChild(li);
        });
    }
    
    clearHistory() {
        this.history.clear();
        this.updateHistory();
    }
    
    showError(message) {
        const errorDiv = document.getElementById('error');
        errorDiv.textContent = message;
        errorDiv.style.display = 'block';
        
        setTimeout(() => {
            errorDiv.style.display = 'none';
        }, 3000);
    }
    
    updateDisplay() {
        const currencies = this.converter.getAvailableCurrencies();
        document.getElementById('availableCurrencies').textContent = 
            `Доступные валюты: ${currencies.join(', ')}`;
    }
}
