Переменные в JavaScript

Начальный уровень 25 минут 7 разделов

Переменные — это контейнеры для хранения данных. Они позволяют сохранять значения для последующего использования в программе. Понимание работы с переменными — основа программирования на JavaScript!

Что такое переменные?

Представьте переменные как коробки с названиями, в которых можно хранить разные значения:

userName
"Анна"
String
userAge
25
Number
isOnline
true
Boolean

Простая аналогия:

Переменные — это как наклейки на коробках в вашем гараже. Вы пишете "Новогодние украшения" на коробке, и всегда знаете, где их найти. Точно так же переменная userName хранит значение "Анна".

Попробуйте сами:

// Создаем переменные
let greeting = "Привет, мир!";
let number = 42;
let isReady = true;

// Используем переменные
console.log(greeting);     // "Привет, мир!"
console.log(number);       // 42
console.log(isReady);      // true

// Меняем значение переменной
number = 100;
console.log(number);       // 100

// Складываем значения
let a = 5;
let b = 10;
let sum = a + b;
console.log(sum);          // 15

Объявление переменных

В JavaScript есть три способа объявления переменных. Каждый имеет свои особенности:

var

Устаревший

Использовался до ES6. Имеет функцию или глобальную область видимости.

var oldVariable = "Устарело";

let

Современный

Блочная область видимости. Можно изменять значение после объявления.

let modernVariable = "Можно менять";

const

Рекомендуемый

Блочная область видимости. Нельзя переопределять (константа).

const constantVariable = "Неизменно";

Сравнение var, let и const

Понимание различий между этими ключевыми словами критически важно:

Критерий var let const
Область видимости Функция или глобальная Блочная Блочная
Можно переопределить ✅ Да ✅ Да ❌ Нет
Можно обновить значение ✅ Да ✅ Да ⚠️ Частично*
Подвержена всплытию ✅ Да (undefined) ✅ Да (ошибка) ✅ Да (ошибка)
Рекомендуется к использованию ❌ Нет ✅ Для изменяемых ✅ Для неизменяемых

Важное замечание про const:

const защищает от переопределения переменной, но не делает объект неизменяемым! 

const user = { name: "Анна", age: 25 };
user.age = 26;                    // ✅ Можно!
console.log(user.age);            // 26

user = { name: "Мария" };         // ❌ Ошибка! Переопределение

Интерактивные примеры:

Различия между let и var

// var - видна вне блока
if (true) {
    var varVariable = "Видна снаружи";
    let letVariable = "Не видна снаружи";
}
console.log(varVariable);    // ✅ "Видна снаружи"
console.log(letVariable);    // ❌ Ошибка! letVariable is not defined

Что можно и нельзя с const

const PI = 3.14159;
PI = 3.14;                     // ❌ Ошибка!

const colors = ["красный", "зеленый"];
colors.push("синий");          // ✅ Можно!
colors = ["желтый"];           // ❌ Ошибка!

const person = { name: "Иван" };
person.name = "Петр";          // ✅ Можно!
person = { name: "Сергей" };   // ❌ Ошибка!

Блочная область видимости

for (let i = 0; i < 3; i++) {
    // i существует только в цикле
    console.log(i);           // 0, 1, 2
}
console.log(i);               // ❌ Ошибка! i is not defined

for (var j = 0; j < 3; j++) {
    // j видна снаружи
}
console.log(j);               // ✅ 3 (видна снаружи!)

Правила именования переменных

JavaScript имеет строгие правила для имен переменных. Их соблюдение делает код чище и понятнее.

Обязательные правила:

  • Имя может содержать буквы, цифры, $ и _
  • Первый символ не может быть цифрой
  • Чувствительны к регистру (age ≠ Age ≠ AGE)
  • Нельзя использовать зарезервированные слова

Хорошие практики:

  • Используйте camelCase для переменных (userName)
  • Используйте UPPER_CASE для констант (MAX_SIZE)
  • Имена должны быть осмысленными
  • Избегайте сокращений (user лучше чем usr)

Плохие практики:

  • Однобуквенные имена (кроме счетчиков)
  • Слишком длинные имена
  • Использование транслита (imya вместо name)
  • Смешение языков в именах

Хорошие примеры:

let userName = "Анна";
const MAX_USERS = 100;
let isLoggedIn = true;
let itemCount = 5;
let currentDate = new Date();

Плохие примеры:

let a = "Анна";            // Что такое "a"?
let imyapolzovatelya = "..." // Транслит
let UserName = "..."         // PascalCase для переменной
let my_super_variable_name_with_long_name = 1 // Слишком длинно

Присваивание и изменение значений

Оператор присваивания (=) сохраняет значение в переменную. Можно использовать составные операторы для краткости.

Оператор Пример Эквивалент Описание
= x = 5 x = 5 Простое присваивание
+= x += 3 x = x + 3 Прибавить и присвоить
-= x -= 2 x = x - 2 Вычесть и присвоить
*= x *= 4 x = x * 4 Умножить и присвоить
/= x /= 2 x = x / 2 Разделить и присвоить
%= x %= 3 x = x % 3 Остаток от деления

Демонстрация изменения значений:

Переменная: counter
Текущее значение: 0 
let counter = 0;

Область видимости (Scope)

Область видимости определяет, где переменная доступна для использования.

Глобальная область

Переменная доступна везде в коде

// Вне всех функций и блоков
let globalVar = "Я везде!";

function test() {
    console.log(globalVar); // ✅ Доступна
}

Локальная (функция)

Доступна только внутри функции

function myFunction() {
    let localVar = "Только здесь";
    console.log(localVar); // ✅
}
console.log(localVar); // ❌ Ошибка!

Блочная (ES6)

Доступна только внутри блока {}

if (true) {
    let blockVar = "В блоке";
    console.log(blockVar); // ✅
}
console.log(blockVar); // ❌ Ошибка!

Визуализация области видимости:

Глобальная область
globalVar
Функция outer()
outerVar
Блок if
blockVar

Всплытие (Hoisting)

JavaScript "поднимает" объявления переменных в начало их области видимости перед выполнением кода.

Как работает hoisting:

Исходный код:
console.log(x); // undefined
var x = 5;
console.log(x); // 5

console.log(y); // Ошибка!
let y = 10;
После hoisting:
var x; // Объявление поднято
console.log(x); // undefined
x = 5; // Присваивание осталось
console.log(x); // 5

// let не инициализируется
console.log(y); // Ошибка!
let y = 10;

Правила hoisting:

  • var: объявление поднимается, инициализируется как undefined
  • let/const: объявление поднимается, но не инициализируется (временная мертвая зона)
  • Функции: полностью поднимаются (можно вызывать до объявления)

Практическое задание

Применим знания на практике:

Задача: Конвертер валют

Средний уровень

Создайте программу для конвертации валют:

  1. Объявите константы для курсов валют (USD, EUR)
  2. Запросите у пользователя сумму в рублях
  3. Вычислите эквивалент в долларах и евро
  4. Выведите результаты с пояснениями
  5. Добавьте возможность выбрать целевую валюту

Подсказки:

  • Используйте const для курсов валют
  • Для ввода используйте prompt() или parseFloat()
  • Форматируйте вывод с помощью toFixed(2)
  • Проверяйте корректность ввода

Редактор кода:

Вывод вашего кода появится здесь

Пример решения:

// Конвертер валют

// 1. Объявляем константы с курсами
const USD_RATE = 90.5;    // 1 USD = 90.5 RUB
const EUR_RATE = 98.3;    // 1 EUR = 98.3 RUB

// 2. Запрашиваем сумму в рублях
let rubles = prompt("Введите сумму в рублях:");

// 3. Проверяем корректность ввода
if (rubles === null || rubles === "") {
    console.log("Отмена или пустой ввод");
} else {
    // Преобразуем строку в число
    rubles = parseFloat(rubles);
    
    if (isNaN(rubles) || rubles <= 0) {
        console.log("Ошибка: введите корректную сумму");
    } else {
        // 4. Вычисляем эквиваленты
        let dollars = rubles / USD_RATE;
        let euros = rubles / EUR_RATE;
        
        // 5. Выводим результаты
        console.log("=== Результаты конвертации ===");
        console.log(`Сумма в рублях: ${rubles} RUB`);
        console.log(`В долларах: ${dollars.toFixed(2)} USD`);
        console.log(`В евро: ${euros.toFixed(2)} EUR`);
        
        // 6. Дополнительно: выбор валюты
        let targetCurrency = prompt("Выберите валюту (USD или EUR):")?.toUpperCase();
        
        if (targetCurrency === "USD") {
            console.log(`Вы выбрали USD: ${dollars.toFixed(2)}`);
        } else if (targetCurrency === "EUR") {
            console.log(`Вы выбрали EUR: ${euros.toFixed(2)}`);
        } else {
            console.log("Выбор не распознан, показаны все валюты");
        }
        
        console.log("Спасибо за использование конвертера!");
    }
}

  • Добавьте больше валют (GBP, JPY, CNY)
  • Создайте функцию для конвертации
  • Добавьте кэширование курсов валют
  • Сделайте красивое форматирование вывода
